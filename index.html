async function compileAllForNotebookLM() { if(!requireToken()) return; const token = getToken(); if (!allFilesCache || allFilesCache.length === 0) return alert("Không có dữ liệu! Hãy bấm nút 'Khởi tạo Data Lõi' trước."); if (!confirm(`🚀 XUẤT TOÀN BỘ WORKSPACE CHO NOTEBOOKLM\n\nHệ thống sẽ gộp TẤT CẢ ${allFilesCache.length} bài viết từ mọi Repo thành 1 siêu tệp (Text thuần).\nQuá trình này cực kỳ nhanh vì sử dụng bộ nhớ nội bộ.\nBạn có muốn bắt đầu?`)) return; const toast = document.getElementById('sync-status-toast'); const toastText = document.getElementById('sync-status-text'); if (toast) { toast.style.display = 'flex'; toast.style.backgroundColor = '#8E44AD'; } document.body.style.cursor = 'wait'; try { // Nhóm bài viết theo từng Repo để Sách có cấu trúc rõ ràng const groupedFiles = {}; allFilesCache.forEach(f => { if (!groupedFiles[f.repoName]) groupedFiles[f.repoName] = []; groupedFiles[f.repoName].push(f); }); let textContent = `SIÊU KHO KIẾN THỨC: VIETNDJ WORKSPACE\nCập nhật: ${new Date().toLocaleString('vi-VN')}\nTổng số bài: ${allFilesCache.length}\n====================================================\n\n`; for (const rName in groupedFiles) { textContent += `\n\n####################################################\n`; textContent += `PHẦN: KHO [${rName.toUpperCase()}]\n`; textContent += `####################################################\n\n`; const sortedFiles = groupedFiles[rName].sort((a, b) => (a.timestamp || 0) - (b.timestamp || 0)); for (let i = 0; i < sortedFiles.length; i++) { const f = sortedFiles[i]; if (toastText) toastText.innerText = `Đang nạp: ${f.name}...`; // Tận dụng fullText đã lưu trong DB (Bỏ qua khâu fetch API làm chậm hệ thống) let textBody = f.fullText || f.preview || "Không có nội dung tĩnh."; textContent += `BÀI VIẾT: ${f.name}\n- Thời gian: ${f.fullDate}\n`; const fileTags = (tagsDataCache && tagsDataCache[`${rName}/${f.fileName}`]) || []; if (fileTags.length > 0) textContent += `- Nhãn (Tags): ${fileTags.join(', ')}\n`; const fileLinks = (linksDataCache && linksDataCache[`${rName}/${f.fileName}`]) || []; if (fileLinks.length > 0) textContent += `- Link tham khảo: ${fileLinks.map(l => l.url).join(' | ')}\n`; textContent += `\n[Nội dung]\n${textBody}\n\n----------------------------------------------------\n\n`; // Giả lập độ trễ siêu nhỏ để giao diện không bị treo (non-blocking UI) await new Promise(r => setTimeout(r, 5)); } } if (toastText) toastText.innerText = "Đang tải Siêu Sách lên GitHub Pages..."; const targetRepo = `${username}.github.io`; const exportFileName = `notebooklm_ALL_WORKSPACE.txt`; const targetSha = await getFileShaSafe(`${username}/${targetRepo}`, exportFileName); const uploadContentBase64 = await encodeBase64UTF8Async(textContent); const uploadRes = await fetch(`https://api.github.com/repos/${username}/${targetRepo}/contents/${exportFileName}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: `🤖 Auto-sync NotebookLM: Đóng gói toàn bộ Workspace`, content: uploadContentBase64, sha: targetSha || undefined }) }); if (!uploadRes.ok) throw new Error("Ghi file lên GitHub thất bại."); if (toastText) toastText.innerText = "✅ Đóng gói hoàn tất!"; const publicUrl = `https://${username}.github.io/${exportFileName}`; try { await navigator.clipboard.writeText(publicUrl); } catch(e){} setTimeout(() => { prompt(`🎉 ĐÃ TẠO SIÊU SÁCH AI THÀNH CÔNG!\n\nLink đã tự động Copy vào Clipboard. Dán URL này vào mục "Website" của NotebookLM để AI đọc toàn bộ hệ thống của bạn:\n\n(Nhấn Ctrl+C / Cmd+C để copy)`, publicUrl); }, 500); } catch (error) { alert(`Lỗi hệ thống: ${error.message}`); } finally { setTimeout(() => { if (toast) { toast.style.display = 'none'; toast.style.backgroundColor = ''; } }, 4000); document.body.style.cursor = 'default'; } }
